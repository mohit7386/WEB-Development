    // Function to calculate the number of digits in a given number
    int countDigits(int num) {
        int count = 0;
        while (num != 0) {
            num /= 10;
            count++;
        }
        return count;
    }
    
    // Function to check if a number is Armstrong or not
    int isArmstrong(int num) {
        int originalNum, remainder, result = 0, n = 0;
    
        // Assigning the value of num to originalNum
        originalNum = num;
    
        // Calculating the number of digits
        n = countDigits(num);
    
        // Calculating result
        while (originalNum != 0) {
            remainder = originalNum % 10;
            result += pow(remainder, n);
            originalNum /= 10;
        }
    
        // Checking if num is Armstrong
        if (result == num)
            return 1; // Returns 1 if num is Armstrong
        else
            return 0; // Returns 0 if num is not Armstrong
    }
    
    int main() {
        int num;
        printf("Enter a number: ");
        scanf("%d", &num);
    
        if (isArmstrong(num))
            printf("%d is an Armstrong number.\n", num);
        else
            printf("%d is not an Armstrong number.\n", num);
    
        return 0;
    }
